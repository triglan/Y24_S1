	// [문제] "STL.cpp"의 소문자를  list<char>에 push_back()으로 저장하라.
	ifstream in{ "STL.cpp" };
	if (not in) return -1;

	vector<char> list{};
	char c;
	
	while (in >> c)
		if(islower(c))	//너 소문자야?
			list.push_back(c);

	sort(list.begin(), list.end()); // 벡터의 정렬, algorithm include
	auto p = unique(list.begin(), list.end());
	list.erase(p, list.end());

	// 출력하라
	for (const char& c : list)
		cout << c;
	cout << endl;


	// 모든 소문자가 다 있나 확인하라. - qyzj
	if (list.size() == 26)
		cout << "[a,z] 다 있음" << endl;
	else
		cout << "없는 문자가 있다." << endl;

	save("STL.cpp");
}
